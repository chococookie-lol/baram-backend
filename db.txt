create table Summoner 
(
    accountId varchar(56),
    profileIconId int,
    revisionDate varchar(30),
    name varchar(100),
    id varchar(63),
    puuid varchar(78) PRIMARY KEY,
    summonerLevel long,
    recentUpdate varchar(30)
);

create table Play (
    puuid varchar(78),
    matchId varchar(30),
    gameEndTimestamp long,
PRIMARY KEY(puuid, matchId));

create table Matches (
    matchId varchar(30) PRIMARY KEY,
    gameCreation long,
    gameDuration long,
    gameEndTimestamp long,
    gameId long,
    gameMode varchar(30),
    gameName varchar(60),
    gameStartTimestamp long,
    gameType varchar(40),
    gameVersion varchar(40),
    mapId int,
    platformId varchar(20),
    queueId int,
    tournamentCode varchar(50)
);

create table Participant (
    matchId varchar(30),
    puuid varchar(78),
    participantNumber int,
    goldEarned long,
    totalMinionsKilled long,
    kills int,
    deaths int, 
    assists int,
    kda float,
    killParticipation float,
    championId int,
    championName varchar(30),
    champLevel int,
    totalDamageDealtToChampions long,
    summoner1Id int,
    summoner2Id int,
    item0 int,
    item1 int,
    item2 int,
    item3 int,
    item4 int,
    item5 int,
    item6 int,
    summonerName varchar(100),
    PRIMARY KEY(matchId, puuid)
);
